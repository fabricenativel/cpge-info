	\begin{algorithm}[H]
		\DontPrintSemicolon
		\caption{Algorithme d'Euclide de calcul du {\sc pgcd}}
		\Entree{$a \in \N, b \in \N$}
		\Sortie{Le {\sc pgcd} de $a$ et de $b$}
		\everypar={\footnotesize \textcolor{gray}{\nl}}
		\Tq{$b \neq 0$}{
			$r \leftarrow $ reste dans la division euclidienne de $a$ par $b$ \;
			$a \leftarrow b$ \;
			$b \leftarrow r$ \;
		}
		\Return $a$
	\end{algorithm}
