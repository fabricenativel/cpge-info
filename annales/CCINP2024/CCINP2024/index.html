        def deplacer_graines(plateau, case):
        a_semer = plateau[case]
        plateau[case] = 0
        # La prochaine position oÃ¹ on va semer
        pos = (case + 1) % 12
        while a_semer != 0:
            plateau[pos] += 1
            a_semer -= 1
            pos = (pos + 1) % 12
            # On ne doit pas semer dans la case initiale
            if (pos == case):
                pos = (pos + 1) % 12
        return (pos-1) % 12
