    {
        if (!is_leaf(g))

            make_edge(writer, *g, *(g->sag));
            make_edge(writer, *g, *(g->sad));
            write_edges(writer, g->sag);
            write_edges(writer, g->sad);
        }
    }

        write_edges(writer, g);
        fprintf(writer, "}\n");
        fclose(writer);
